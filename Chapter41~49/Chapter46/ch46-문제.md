### 다음 코드의 실행 결과를 쓰시오
```js
function* genFunc(){
  const x = yield 1;
  const y = yield(x + 1);
  return x + y;
}

const generator = genFunc(0);

let res = generator.next(1);
console.log(res);

res = generator.next(2);
console.log(res);

res = generator.next(10);
console.log(res);
```
