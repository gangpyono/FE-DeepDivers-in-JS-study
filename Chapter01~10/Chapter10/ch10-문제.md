## 📌문제2
### 다음 코드의 출력결과를 작성하시오.
```js
const obj = {
  "abc": 123,
  abc : 789,
  000 : "zero",
  "000" : "stringZero",
}
const ABC = "abc";
const zero = 000;

console.log(obj["abc"]); // (1)
console.log(obj[ABC]); // (2)
console.log(obj.abc); // (3)

console.log(obj[000]); // (4)
console.log(obj["000"]); // (5)
console.log(obj[zero]); // (6)
console.log(obj.zero); // (7)
```
